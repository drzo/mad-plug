# x86-64 Architecture Profile

Target: AMD64/Intel 64 processors with SSE4.1+ / AVX / AVX2 / AVX-512

## Hardware Characteristics

| Feature | SSE4.1 | AVX | AVX2 | AVX-512 |
|---------|--------|-----|------|---------|
| Vector width | 128-bit | 256-bit | 256-bit | 512-bit |
| Registers | 16 × XMM | 16 × YMM | 16 × YMM | 32 × ZMM |
| FMA | No | Optional | Yes | Yes |
| Int8 dot | No | No | No | VNNI |

## Feature Detection

```c
#include <cpuid.h>

bool has_avx2(void) {
    unsigned int eax, ebx, ecx, edx;
    __cpuid_count(7, 0, eax, ebx, ecx, edx);
    return (ebx & (1 << 5)) != 0;  // AVX2 bit
}

bool has_avx512f(void) {
    unsigned int eax, ebx, ecx, edx;
    __cpuid_count(7, 0, eax, ebx, ecx, edx);
    return (ebx & (1 << 16)) != 0;  // AVX-512F bit
}

bool has_f16c(void) {
    unsigned int eax, ebx, ecx, edx;
    __cpuid(1, eax, ebx, ecx, edx);
    return (ecx & (1 << 29)) != 0;  // F16C bit
}
```

### Essential Operations

```c
// Load/Store (AVX2)
__m256 _mm256_loadu_ps(const float* ptr);
void _mm256_storeu_ps(float* ptr, __m256 val);

// Arithmetic
__m256 _mm256_add_ps(__m256 a, __m256 b);
__m256 _mm256_mul_ps(__m256 a, __m256 b);
__m256 _mm256_fmadd_ps(__m256 a, __m256 b, __m256 c);  // a*b + c

// Broadcast
__m256 _mm256_set1_ps(float val);

// Horizontal sum
float hsum_float_8(__m256 x) {
    __m128 hi = _mm256_extractf128_ps(x, 1);
    __m128 lo = _mm256_castps256_ps128(x);
    lo = _mm_add_ps(lo, hi);
    hi = _mm_movehl_ps(lo, lo);
    lo = _mm_add_ps(lo, hi);
    hi = _mm_shuffle_ps(lo, lo, 1);
    lo = _mm_add_ss(lo, hi);
    return _mm_cvtss_f32(lo);
}

// Shuffle/Permute
__m256i _mm256_shuffle_epi8(__m256i a, __m256i b);
__m256 _mm256_permutevar8x32_ps(__m256 a, __m256i idx);
```

## Optimized Patterns

### Vector Dot Product (AVX2)

```c
float vec_dot_f32_avx2(int n, const float* x, const float* y) {
    __m256 sum0 = _mm256_setzero_ps();
    __m256 sum1 = _mm256_setzero_ps();
    
    for (int i = 0; i < n; i += 16) {
        __m256 x0 = _mm256_loadu_ps(x + i);
        __m256 x1 = _mm256_loadu_ps(x + i + 8);
        __m256 y0 = _mm256_loadu_ps(y + i);
        __m256 y1 = _mm256_loadu_ps(y + i + 8);
        
        sum0 = _mm256_fmadd_ps(x0, y0, sum0);
        sum1 = _mm256_fmadd_ps(x1, y1, sum1);
    }
    
    sum0 = _mm256_add_ps(sum0, sum1);
    return hsum_float_8(sum0);
}
```

### Q4_0 Dequantization (AVX2)

```c
void dequantize_row_q4_0_avx2(const block_q4_0* x, float* y, int64_t k) {
    const int nb = k / QK4_0;
    
    for (int i = 0; i < nb; i++) {
        const float d = ggml_fp16_to_fp32(x[i].d);
        const __m256 vd = _mm256_set1_ps(d);
        
        // Load 16 bytes = 32 4-bit values
        const __m128i quants = _mm_loadu_si128((const __m128i*)x[i].qs);
        
        // Expand to 256-bit
        const __m256i quants256 = _mm256_cvtepu8_epi16(quants);
        
        // Extract low and high nibbles
        const __m256i mask = _mm256_set1_epi16(0x000F);
        const __m256i lo = _mm256_and_si256(quants256, mask);
        const __m256i hi = _mm256_srli_epi16(quants256, 4);
        
        // Subtract 8 (symmetric quantization)
        const __m256i offset = _mm256_set1_epi16(8);
        const __m256i lo_s = _mm256_sub_epi16(lo, offset);
        const __m256i hi_s = _mm256_sub_epi16(hi, offset);
        
        // Convert to float and scale
        // ... (widen to 32-bit, cvt to float, multiply by vd)
        
        _mm256_storeu_ps(y + i*32 +  0, ...);
        // ... store all 32 values
    }
}
```

### Q4_0 × Q8_0 Dot Product (AVX2)

```c
void ggml_vec_dot_q4_0_q8_0_avx2(int n, float* s,
                                  const void* vx, const void* vy) {
    const block_q4_0* x = vx;
    const block_q8_0* y = vy;
    
    __m256 acc = _mm256_setzero_ps();
    
    for (int i = 0; i < n/QK4_0; i++) {
        // Load scales
        const float d = ggml_fp16_to_fp32(x[i].d) * ggml_fp16_to_fp32(y[i].d);
        
        // Load x (4-bit packed)
        const __m128i qx = _mm_loadu_si128((const __m128i*)x[i].qs);
        
        // Load y (8-bit)
        const __m256i qy = _mm256_loadu_si256((const __m256i*)y[i].qs);
        
        // Unpack x: split into low/high nibbles, subtract 8
        const __m128i mask = _mm_set1_epi8(0x0F);
        const __m128i qx_lo = _mm_and_si128(qx, mask);
        const __m128i qx_hi = _mm_and_si128(_mm_srli_epi16(qx, 4), mask);
        
        // Interleave and convert to signed
        __m256i qx_256 = _mm256_set_m128i(qx_hi, qx_lo);
        qx_256 = _mm256_sub_epi8(qx_256, _mm256_set1_epi8(8));
        
        // Compute dot product using _mm256_maddubs_epi16
        // (multiply unsigned×signed, add adjacent pairs)
        __m256i prod = _mm256_maddubs_epi16(
            _mm256_abs_epi8(qx_256),  // make unsigned
            _mm256_sign_epi8(qy, qx_256)  // adjust sign
        );
        
        // Horizontal sum and accumulate
        prod = _mm256_madd_epi16(prod, _mm256_set1_epi16(1));
        __m256 prodf = _mm256_cvtepi32_ps(prod);
        acc = _mm256_fmadd_ps(_mm256_set1_ps(d), prodf, acc);
    }
    
    *s = hsum_float_8(acc);
}
```

### CMake Flags

```cmake
# Auto-detect best SIMD
cmake -DGGML_NATIVE=ON ..

# Force specific level
cmake -DGGML_AVX=ON -DGGML_AVX2=OFF ..

cmake -DGGML_AVX512=ON ..
```

## CPU-Specific Optimizations

### Intel

| Generation | Features | Flags |
|------------|----------|-------|
| Sandy Bridge | AVX | `-mavx` |
| Haswell | AVX2, FMA | `-mavx2 -mfma` |
| Skylake-X | AVX-512 | `-mavx512f -mavx512bw` |
| Ice Lake | AVX-512 VNNI | `-mavx512vnni` |
| Sapphire Rapids | AMX, BF16 | `-mamx-tile -mamx-int8` |

### AMD

| Generation | Features | Flags |
|------------|----------|-------|
| Zen | AVX2 | `-march=znver1` |
| Zen 2 | AVX2 (wider) | `-march=znver2` |
| Zen 3 | AVX2 | `-march=znver3` |
| Zen 4 | AVX-512, VNNI | `-march=znver4` |

## Performance Considerations

| Operation | SSE4.1 | AVX | AVX2 | AVX-512 |
|-----------|--------|-----|------|---------|
| F32 FLOPS/cycle | 8 | 16 | 16 | 32 |
| Memory BW util | Good | Good | Good | May throttle |
| Power efficiency | Best | Good | Good | Varies |

**AVX-512 Caveats:**
- May cause frequency throttling on some CPUs
- Not always faster than AVX2 for small workloads
- Check `GGML_AVX512` performance on target hardware
